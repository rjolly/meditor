// Requires : org.apache.commons#commons-math3;3.5

import scala.language.implicitConversions
import math3.{Matrix, int2matrix}
import scas.int2bigInt

given m: Matrix = new Matrix(3)

val a = m(
  0, 1, 0,
  0, 0, 1,
  1, 0, 0
)
val b = m(
  0, 0, 1,
  1, 0, 0,
  0, 1, 0
)
assert(a + 1 >< 1 + a)
assert(a%* 2 >< 2 *%a)
assert(a%/ 2 >< .5*%a)
assert(a \ -1 >< b)
assert(a * b >< 1)
